# 1️⃣ Base image: Python 3.11 slim
FROM python:3.11-slim

# 2️⃣ Set working directory inside container
WORKDIR /app

# 3️⃣ Copy requirements.txt
COPY requirements.txt .

# 4️⃣ Upgrade pip and install dependencies
RUN pip install --upgrade pip
RUN pip install -r requirements.txt

# 5️⃣ Copy the rest of your project files
COPY . .

# 6️⃣ Expose port 10000 (Langflow default)
EXPOSE 10000

# 7️⃣ Command to run Langflow
CMD ["sh", "-c", "langflow run --host 0.0.0.0 --port ${PORT:-10000}"]
